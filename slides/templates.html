A little bit of background

                

                


                    
Underscore.js is a utility belt for JavaScript.

                    
It's very well documented.

                    
It's also one of the best examples of the kind of code that needs to be shared.

                    
Made by DocumentCloud - same people who make Backbone.js.

                

                

            


            

                
Using Underscore

                

                
After you install underscore...

                    
var _ = require('underscore');
                    

                    
Because _ is a valid character, we use it by itself. Now we have access to some _ namespaced methods.

                    
_.each([1, 2, 3], alert);

var stooges = [{name : 'moe', age : 40}, {name : 'larry', age : 50}, {name : 'curly', age : 60}];
_.pluck(stooges, 'name');
=> ["moe", "larry", "curly"]

_(3).times(function(n){ genie.grantWishNumber(n); });

_.escape('Curly, Larry & Moe');
                    

                

            


            

                
Using Templates

                

                    
var _ = require('underscore');

var compiled = _.template("hello: <%= name %>");
compiled({name : 'moe'});
=> "hello: moe"
                    

                    
Or...

                    
var _ = require('underscore');

var compiled = _.template("Goodbye, <%= name %>",{name : 'Joe'});
=> "Goodbye, Joe"
                    

                

                
                    
Note, we're using the underscore character in a similar way that jQuery is using the dollar sign. It's like namespacing.
